7.3: Script de TDP unlock
bashnano /usr/local/bin/tdp-unlock.sh
Conteúdo:
bash#!/bin/bash

if [ "$EUID" -ne 0 ]; then 
    echo "Execute como root"
    exit 1
fi

modprobe msr

# PL1=20W, PL2=28W (conservador)
wrmsr -a 0x610 0x00e08004e08000

# Turbo all-core: 3.2 GHz
wrmsr -a 0x1ad 0x2020202020202020

# Time window infinito
wrmsr -a 0x618 0x0

echo "TDP unlocked: PL1=20W, PL2=28W, Turbo=3.2GHz"
Salvo e torno executável:
bashchmod +x /usr/local/bin/tdp-unlock.sh
7.4: Systemd service

[Install]
WantedBy=multi-user.target
Salvo e habilito:

📊 PASSO 3: SCRIPT DE MONITORAMENTO
bashnano /usr/local/bin/monitor-system.sh
Conteúdo:
bash#!/bin/bash

while true; do
    clear
    echo "╔════════════════════════════════════════════════════════════╗"
    echo "║         DEBIAN TUNADO 3.0 - MONITOR REALTIME             ║"
    echo "╚════════════════════════════════════════════════════════════╝"
    echo ""
    
    # CPU
    TEMP=$(sensors 2>/dev/null | grep "Core 0" | awk '{print $3}' | tr -d '+°C' || echo "N/A")
    FREQ=$(cat /sys/devices/system/cpu/cpu0/cpufreq/scaling_cur_freq)
    FREQ_GHZ=$(echo "scale=2; $FREQ / 1000000" | bc)
    
    echo "┌─ CPU ──────────────────────────────────────────────────────┐"
    echo "│ Freq: ${FREQ_GHZ} GHz | Temp: ${TEMP}°C"
    echo "└────────────────────────────────────────────────────────────┘"
    echo ""
    
    # Memória
    echo "┌─ MEMÓRIA ──────────────────────────────────────────────────┐"
    free -h | grep -E "Mem|Swap" | awk '{printf "│ %-6s %6s / %6s (usado: %6s)\n", $1, $3, $2, $3}'
    echo "└────────────────────────────────────────────────────────────┘"
    echo ""
    
    # Swap layers
    echo "┌─ SWAP LAYERS ──────────────────────────────────────────────┐"
    swapon --show --bytes 2>/dev/null | awk 'NR>1 {printf "│ %-20s %8.1f MB (pri: %3s)\n", $1, $3/1024/1024, $5}'
    echo "└────────────────────────────────────────────────────────────┘"
    echo ""
    
    # Docker
    if systemctl is-active docker &> /dev/null; then
        echo "┌─ DOCKER ───────────────────────────────────────────────────┐"
        CONTAINERS=$(docker ps -q 2>/dev/null | wc -l)
        echo "│ Containers ativos: ${CONTAINERS}"
        if [ "$CONTAINERS" -gt 0 ]; then
            docker stats --no-stream --format "│ {{.Name}}: CPU {{.CPUPerc}} | MEM {{.MemUsage}}" 2>/dev/null
        fi
        echo "└────────────────────────────────────────────────────────────┘"
    fi
    
    sleep 2
done

bash#!/bin/bash

DATE=$(date +%Y%m%d-%H%M)

# Cleanup logs
journalctl --vacuum-time=7d
