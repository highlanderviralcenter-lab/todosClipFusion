#!/bin/bash

[ "$EUID" -ne 0 ] && echo 'Execute como root' && exit 1

modprobe msr

# i5-6200U: PL1=20W sustentado, PL2=25W burst
# Com refrigeração caseira: 20W é seguro e eficiente
wrmsr -a 0x610 0x00dc8004dc8000
  # PL1=20W sustentado, PL2=25W burst

# Turbo all-core: 2.7GHz (27x = 0x1B) — ponto ótimo de eficiência
# 2.8GHz (0x1C) dá só +3% mas consome muito mais — não vale
wrmsr -a 0x1ad 0x1B1B1B1B1B1B1B1B

# Time window: 28s (padrão Intel)
wrmsr -a 0x618 0x0

echo "TDP unlocked: PL1=20W, PL2=25W, Turbo=2.7GHz (i5-6200U)"

chmod +x /usr/local/bin/tdp-unlock.sh

echo ""
echo "═══ 2. KERNEL PARAMS ═══"
check_param() {
  echo -n "$1: "
  grep -q "$2" /proc/cmdline && echo -e "${GREEN}✅${NC}" || { echo -e "${RED}❌${NC}"; ((FAILED++)); }
}
check_param 'i915.enable_guc=3' 'i915.enable_guc=3'

  echo -e "${GREEN}✅${NC}"
else
  echo -e "${RED}❌${NC}"; ((FAILED++))

echo ""
echo "═══ 4. VA-API ═══"
