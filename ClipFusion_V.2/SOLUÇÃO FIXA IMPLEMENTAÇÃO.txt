#!/usr/bin/env python3
"""
db_bootstrap.py - Unificação do Schema de Dados ClipFusion

Resolve o Schema Mismatch entre viral_analyzer e prompt_builder.
Garante consistência nos campos: hook_score, retention_score, moment_score
"""

import sqlite3
import os
from pathlib import Path

class DatabaseBootstrap:
    SCHEMA_V2 = """
    -- Tabela principal de projetos
    CREATE TABLE IF NOT EXISTS projetos (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        nome TEXT NOT NULL,
        video_path TEXT NOT NULL,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        status TEXT DEFAULT 'pending'
    );
    
    -- Tabela de cortes com schema unificado (VERSÃO CORRIGIDA)
    CREATE TABLE IF NOT EXISTS cortes (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        projeto_id INTEGER REFERENCES projetos(id),
        
        -- Identificação do corte
        start_time TEXT NOT NULL,
        end_time TEXT NOT NULL,
        duration_seconds REAL,
        
        -- Scores unificados (nomes padronizados)
        hook_score REAL DEFAULT 0.0,           -- Era 'gancho' ou 'hook'
        retention_score REAL DEFAULT 0.0,      -- Era 'retencao_estimada' ou 'retention'
        moment_score REAL DEFAULT 0.0,         -- Era 'momento' ou 'moment'
        
        -- Score composto (Regra de Ouro)
        viral_score REAL DEFAULT 0.0,          -- Nota final ≥ 9.0 para aprovação
        
        -- Metadados
        archetype TEXT,                        -- Arquétipo emocional detectado
        platform TEXT,                         -- TikTok/Reels/Shorts
        approved BOOLEAN DEFAULT 0,            -- Aprovado para render
        
        -- Conteúdo
        transcription TEXT,
        hook_text TEXT,                        -- Texto do gancho gerado
        
        -- Status
        rendered BOOLEAN DEFAULT 0,
        output_path TEXT,
        
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    );
    
    -- Índices para performance
    CREATE INDEX IF NOT EXISTS idx_cortes_projeto ON cortes(projeto_id);
    CREATE INDEX IF NOT EXISTS idx_cortes_score ON cortes(viral_score);
    CREATE INDEX IF NOT EXISTS idx_cortes_approved ON cortes(approved);
    
    -- Tabela de configurações de hardware
    CREATE TABLE IF NOT EXISTS hardware_config (
        id INTEGER PRIMARY KEY,
        zram_enabled BOOLEAN DEFAULT 0,
        vaapi_enabled BOOLEAN DEFAULT 0,
        gpu_acceleration BOOLEAN DEFAULT 0,
        last_updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    );
    """
    
    def __init__(self, db_path="clipfusion.db"):
        self.db_path = db_path
        self.conn = None
        
    def init_database(self):
        """Inicializa o banco com schema unificado"""
        self.conn = sqlite3.connect(self.db_path)
        self.conn.executescript(self.SCHEMA_V2)
        self.conn.commit()
        print(f"[DB_BOOTSTRAP] Banco inicializado: {self.db_path}")
        return self.conn
    
    def migrate_from_legacy(self, old_db_path=None):
        """
        Migra dados de schema antigo para novo sem perder projetos.
        Converte: retencao_estimada → retention_score
        """
        if old_db_path and os.path.exists(old_db_path):
            old_conn = sqlite3.connect(old_db_path)
            old_cursor = old_conn.cursor()
            
            try:
                # Tenta ler dados antigos
                old_cursor.execute("""
                    SELECT start_time, end_time, retencao_estimada, 
                           gancho, momento, approved 
                    FROM cortes_old
                """)
                
                rows = old_cursor.fetchall()
                print(f"[MIGRATION] Migrando {len(rows)} cortes do schema antigo...")
                
                for row in rows:
                    start, end, retencao, gancho, momento, approved = row
                    # Converte nomes antigos para novos
                    self.conn.execute("""
                        INSERT INTO cortes 
                        (start_time, end_time, retention_score, hook_score, 
                         moment_score, approved, viral_score)
                        VALUES (?, ?, ?, ?, ?, ?, ?)
                    """, (start, end, retencao, gancho, momento, approved,
                          (retencao + gancho + momento) / 3))
                
                self.conn.commit()
                print("[MIGRATION] Migração concluída com sucesso")
                
            except sqlite3.OperationalError as e:
                print(f"[MIGRATION] Schema antigo não encontrado ou já migrado: {e}")
            finally:
                old_conn.close()
    
    def validate_schema(self):
        """Valida se o schema está correto"""
        cursor = self.conn.cursor()
        cursor.execute("PRAGMA table_info(cortes)")
        columns = {row[1] for row in cursor.fetchall()}
        
        required = {'hook_score', 'retention_score', 'moment_score', 'viral_score'}
        
        if required.issubset(columns):
            print("[VALIDATION] Schema validado com sucesso")
            return True
        else:
            missing = required - columns
            print(f"[VALIDATION] ERRO: Colunas faltando: {missing}")
            return False
    
    def get_connection(self):
        """Retorna conexão ativa"""
        if not self.conn:
            self.init_database()
        return self.conn

# Singleton para uso global
_db_instance = None

def get_db(db_path="clipfusion.db"):
    global _db_instance
    if _db_instance is None:
        _db_instance = DatabaseBootstrap(db_path)
        _db_instance.init_database()
    return _db_instance

if __name__ == "__main__":
    # Teste standalone
    bootstrap = DatabaseBootstrap()
    bootstrap.init_database()
    bootstrap.validate_schema()

#!/usr/bin/env python3
"""
decision_engine.py - Motor de Decisão SignalCut Hybrid

Implementa a Regra de Ouro: Nota ≥ 9.0 para aprovação automática
Unifica pontuação local com análise da IA externa
"""

import json
import sqlite3
from typing import List, Dict, Optional
from dataclasses import dataclass

@dataclass
class CutDecision:
    cut_id: int
    start_time: str
    end_time: str
    hook_score: float
    retention_score: float
    moment_score: float
    viral_score: float
    approved: bool
    reason: str

class DecisionEngine:
    """
    Motor de decisão que implementa a filosofia:
    Fluxo Local → IA Externa → Veredito Local (Regra de Ouro)
    """
    
    GOLDEN_RULE_THRESHOLD = 9.0
    
    # Pesos para cálculo do viral_score
    WEIGHTS = {
        'hook': 0.4,      # Gancho inicial é 40%
        'retention': 0.35, # Retenção é 35%
        'moment': 0.25     # Momento é 25%
    }
    
    def __init__(self, db_connection):
        self.conn = db_connection
        self.cursor = db_connection.cursor()
    
    def calculate_viral_score(self, hook: float, retention: float, moment: float) -> float:
        """
        Calcula score viral ponderado (0-10)
        """
        score = (
            hook * self.WEIGHTS['hook'] +
            retention * self.WEIGHTS['retention'] +
            moment * self.WEIGHTS['moment']
        ) * 10  # Normaliza para escala 0-10
        
        return round(min(10.0, max(0.0, score)), 2)
    
    def apply_golden_rule(self, viral_score: float) -> tuple[bool, str]:
        """
        Aplica Regra de Ouro: Nota ≥ 9.0 = Aprovação automática
        """
        if viral_score >= self.GOLDEN_RULE_THRESHOLD:
            return True, f"Nota {viral_score} ≥ 9.0 - Aprovado automaticamente"
        elif viral_score >= 7.5:
            return False, f"Nota {viral_score} - Revisão recomendada (potencial alto)"
        else:
            return False, f"Nota {viral_score} - Reprovado (abaixo do threshold)"
    
    def process_ia_response(self, projeto_id: int, ia_json: str) -> List[CutDecision]:
        """
        Processa resposta da IA externa e aplica veredito local
        
        Args:
            projeto_id: ID do projeto
            ia_json: JSON retornado pela IA (Claude/ChatGPT)
        
        Returns:
            Lista de decisões de cortes
        """
        try:
            ia_data = json.loads(ia_json)
            cortes_ia = ia_data.get('cortes', ia_data if isinstance(ia_data, list) else [])
        except json.JSONDecodeError:
            print("[DECISION_ENGINE] ERRO: JSON inválido da IA")
            return []
        
        decisions = []
        
        for corte in cortes_ia:
            # Extrai scores da IA (suporta múltiplos nomes de campos)
            hook = corte.get('hook_score', corte.get('gancho', corte.get('hook', 0)))
            retention = corte.get('retention_score', corte.get('retencao_estimada', 
                                                              corte.get('retention', 0)))
            moment = corte.get('moment_score', corte.get('momento', corte.get('moment', 0)))
            
            # Calcula score viral final
            viral = self.calculate_viral_score(hook, retention, moment)
            
            # Aplica Regra de Ouro
            approved, reason = self.apply_golden_rule(viral)
            
            decision = CutDecision(
                cut_id=None,  # Será definido após inserção no DB
                start_time=corte.get('inicio', corte.get('start', '00:00:00')),
                end_time=corte.get('fim', corte.get('end', '00:00:00')),
                hook_score=hook,
                retention_score=retention,
                moment_score=moment,
                viral_score=viral,
                approved=approved,
                reason=reason
            )
            
            # Persiste no banco
            decision.cut_id = self._persist_cut(projeto_id, decision, corte)
            decisions.append(decision)
            
            print(f"[DECISION] Corte {decision.start_time}-{decision.end_time}: "
                  f"Nota {viral} - {'APROVADO' if approved else 'REPROVADO'}")
        
        return decisions
    
    def _persist_cut(self, projeto_id: int, decision: CutDecision, raw_data: dict) -> int:
        """Persiste corte no banco com schema unificado"""
        self.cursor.execute("""
            INSERT INTO cortes 
            (projeto_id, start_time, end_time, duration_seconds,
             hook_score, retention_score, moment_score, viral_score,
             approved, archetype, platform, transcription, hook_text)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        """, (
            projeto_id,
            decision.start_time,
            decision.end_time,
            raw_data.get('duracao', 30),
            decision.hook_score,
            decision.retention_score,
            decision.moment_score,
            decision.viral_score,
            decision.approved,
            raw_data.get('arquetipo', 'unknown'),
            raw_data.get('plataforma', 'tiktok'),
            raw_data.get('transcricao', ''),
            raw_data.get('texto_hook', '')
        ))
        self.conn.commit()
        return self.cursor.lastrowid
    
    def get_approved_cuts(self, projeto_id: int) -> List[Dict]:
        """Retorna cortes aprovados pela Regra de Ouro para renderização"""
        self.cursor.execute("""
            SELECT id, start_time, end_time, duration_seconds,
                   hook_text, archetype, viral_score
            FROM cortes 
            WHERE projeto_id = ? AND approved = 1 AND rendered = 0
            ORDER BY viral_score DESC
        """, (projeto_id,))
        
        columns = [description[0] for description in self.cursor.description]
        return [dict(zip(columns, row)) for row in self.cursor.fetchall()]
    
    def mark_as_rendered(self, cut_id: int, output_path: str):
        """Marca corte como renderizado"""
        self.cursor.execute("""
            UPDATE cortes SET rendered = 1, output_path = ? WHERE id = ?
        """, (output_path, cut_id))
        self.conn.commit()

#!/usr/bin/env python3
"""
cut_engine.py - Render 2-Pass SignalCut Hybrid

Passo 1: VA-API (hardware) para corte + filtros anti-copy
Passo 2: libx264 (software) para queima de legendas
"""

import os
import subprocess
import tempfile
from pathlib import Path
from protection_factory import ProtectionFactory  # 7 módulos anti-copy

class CutEngine:
    """
    Engine de renderização 2-pass otimizado para i5-6200U + Intel HD 520
    """
    
    def __init__(self):
        self.protection = ProtectionFactory()
        self.use_vaapi = self._check_vaapi()
        self.temp_dir = tempfile.mkdtemp(prefix="clipfusion_")
        
    def _check_vaapi(self) -> bool:
        """Verifica se VA-API está disponível"""
        try:
            result = subprocess.run(
                ["vainfo"], capture_output=True, text=True, check=True
            )
            return "VAProfileH264" in result.stdout
        except:
            print("[CUT_ENGINE] VA-API não disponível, usando software")
            return False
    
    def render_cut(self, video_path: str, start_time: str, end_time: str, 
                   cut_id: int, protection_level: str = 'anti_ai') -> str:
        """
        Renderiza um corte usando técnica 2-pass
        
        Args:
            video_path: Caminho do vídeo fonte
            start_time: Tempo inicial (HH:MM:SS)
            end_time: Tempo final (HH:MM:SS)
            cut_id: ID do corte no banco
            protection_level: none, basic, anti_ai, maximum
        
        Returns:
            Caminho do arquivo renderizado
        """
        duration = self._calc_duration(start_time, end_time)
        
        # Arquivos temporários
        temp_cut = os.path.join(self.temp_dir, f"cut_{cut_id}_raw.mp4")
        final_output = os.path.join("output", f"viral_cut_{cut_id}_{protection_level}.mp4")
        
        os.makedirs("output", exist_ok=True)
        
        # PASSO 1: Corte + Proteção Anti-Copy via VA-API (hardware)
        print(f"[PASS 1] Corte com VA-API + Proteção {protection_level}...")
        self._pass1_cut_and_protect(
            video_path, start_time, duration, temp_cut, protection_level
        )
        
        # PASSO 2: Queima de legendas via libx264 (software)
        print(f"[PASS 2] Queima de legendas...")
        self._pass2_burn_subtitles(temp_cut, final_output)
        
        # Limpeza
        if os.path.exists(temp_cut):
            os.remove(temp_cut)
        
        return final_output
    
    def _pass1_cut_and_protect(self, input_path: str, start: str, duration: int,
                                output: str, level: str):
        """Passo 1: Corte com filtros de proteção via VA-API"""
        
        # Gera filtros de proteção
        filters = self.protection.get_ffmpeg_filters(level)
        
        cmd = [
            "ffmpeg",
            "-hwaccel", "vaapi",
            "-vaapi_device", "/dev/dri/renderD128",
            "-ss", start,
            "-t", str(duration),
            "-i", input_path,
            "-vf", f"{filters},format=nv12|vaapi,hwupload",
            "-c:v", "h264_vaapi",
            "-qp", "23",
            "-c:a", "aac",
            "-b:a", "128k",
            "-y", output
        ]
        
        subprocess.run(cmd, check=True, capture_output=True)
    
    def _pass2_burn_subtitles(self, input_path: str, output: str):
        """Passo 2: Queima de legendas (se existirem) via software"""
        
        # Verifica se existe arquivo de legendas
        srt_path = input_path.replace(".mp4", ".srt")
        
        if os.path.exists(srt_path):
            filter_complex = f"subtitles={srt_path}"
        else:
            filter_complex = "copy"
        
        cmd = [
            "ffmpeg",
            "-i", input_path,
            "-vf", filter_complex,
            "-c:v", "libx264",
            "-preset", "fast",
            "-crf", "23",
            "-c:a", "copy",
            "-movflags", "+faststart",
            "-y", output
        ]
        
        subprocess.run(cmd, check=True, capture_output=True)
    
    def _calc_duration(self, start: str, end: str) -> int:
        """Calcula duração em segundos"""
        def to_seconds(t):
            h, m, s = map(int, t.split(':'))
            return h * 3600 + m * 60 + s
        
        return to_seconds(end) - to_seconds(start)

    print(f"VA-API disponível: {engine.use_vaapi}")

#!/usr/bin/env python3
"""
protection_factory.py - Fábrica de Proteção Anti-Copy

Implementa as 7 técnicas de blindagem:
1. Zoom dinâmico (1-3%)
2. Ajuste de cor (matiz/saturation)
3. Ruído sutil (noise)
4. Chroma shift
5. Frequência de áudio (pitch ±0.05st)
6. Time stretch (±0.5%)
7. Perturbação de embeddings (anti-IA)
"""

import random

class ProtectionFactory:
    """
    Fábrica que gera filtros FFmpeg para proteção anti-copyright
    """
    
    LEVELS = {
        'none': [],
        'basic': ['zoom', 'color', 'metadata'],
        'anti_ai': ['zoom', 'color', 'noise', 'chroma', 'audio_pitch', 'temporal'],
        'maximum': ['zoom', 'color', 'noise', 'chroma', 'audio_pitch', 
                   'temporal', 'ai_evasion', 'fingerprint']
    }
    
    def __init__(self):
        self.zoom_counter = 0
    
    def get_ffmpeg_filters(self, level: str) -> str:
        """
        Gera string de filtros FFmpeg para o nível especificado
        """
        techniques = self.LEVELS.get(level, self.LEVELS['basic'])
        filters = []
        
        if 'zoom' in techniques:
            # Zoom dinâmico 1-3% com movimento sutil
            zoom = 1.0 + (random.random() * 0.02 + 0.01)  # 1.01 a 1.03
            x = f"(iw-{zoom}*iw)/2"
            y = f"(ih-{zoom}*ih)/2"
            filters.append(f"zoompan=z='min(zoom+0.001, {zoom})':x={x}:y={y}:d=1")
        
        if 'color' in techniques:
            # Ajuste sutil de matiz e saturação
            hue = random.uniform(-5, 5)  # ±5 graus
            sat = random.uniform(0.95, 1.05)  # ±5% saturação
            filters.append(f"hue=h={hue}:s={sat}")
        
        if 'noise' in techniques:
            # Ruído gaussiano sutil
            filters.append("noise=alls=2:allf=t")
        
        if 'chroma' in techniques:
            # Shift de chroma sutil
            cr = random.uniform(-2, 2)
            cb = random.uniform(-2, 2)
            filters.append(f"eq=cr={cr}:cb={cb}")
        
        if 'temporal' in techniques:
            # Variação de velocidade ±0.5%
            speed = random.uniform(0.995, 1.005)
            filters.append(f"setpts=PTS/{speed}")
        
        # Filtros de áudio
        audio_filters = []
        if 'audio_pitch' in techniques:
            # Pitch shift ±0.05 semitons
            pitch = random.uniform(-0.05, 0.05)
            audio_filters.append(f"rubberband=pitch={pitch}")
        
        # Combina filtros
        video_chain = ",".join(filters) if filters else "copy"
        
        return video_chain
    
    def get_audio_filters(self, level: str) -> str:
        """Retorna filtros de áudio específicos"""
        techniques = self.LEVELS.get(level, [])
        filters = []
        
        if 'audio_pitch' in techniques:
            pitch = random.uniform(-0.05, 0.05)
            filters.append(f"asetrate=48000*{1+pitch/100},aresample=48000")
        
        if 'temporal' in techniques:
            speed = random.uniform(0.995, 1.005)
            filters.append(f"atempo={speed}")
        
        return ",".join(filters) if filters else "anull"

if __name__ == "__main__":
    factory = ProtectionFactory()
    for level in ['none', 'basic', 'anti_ai', 'maximum']:
        print(f"{level}: {factory.get_ffmpeg_filters(level)}")
