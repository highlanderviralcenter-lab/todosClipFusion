import sys
from faster_whisper import WhisperModel

# --- Configuração (ajuste o caminho do vídeo) ---
video_path = "/caminho/para/seu/video_de_teste.mp4"  # MUDE AQUI!
model_size = "tiny"

# --- Inicializa o modelo ---
print(f"Carregando modelo {model_size}...")
model = WhisperModel(model_size, device="cpu", compute_type="int8")

# --- Faz a transcrição ---
print("Iniciando transcrição...")
segments, info = model.transcribe(
    video_path,
    language="pt",
    beam_size=3,
    best_of=3,
    vad_filter=True,
    vad_parameters=dict(min_silence_duration_ms=500)
)

# --- Mostra o resultado ---
print(f"Idioma detectado: {info.language}\n")
for segment in segments:
    print(f"[{segment.start:.2f}s -> {segment.end:.2f}s] {segment.text}")

Edite a linha video_path para apontar para um vídeo curto (30s) que você tenha.

Rode o teste:

Se isso funcionar, você tem a prova de que o faster-whisper está configurado e roda no seu hardware. Se falhar, o erro será simples e vamos corrigir.
