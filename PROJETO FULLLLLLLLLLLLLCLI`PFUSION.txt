// Backend Service Example
class ContentGenerationService {
  
  async generate(request: GenerationRequest) {
    // 1. Valida e registra no PostgreSQL
    const generation = await this.postgres.createGeneration({
      userId: request.userId,
      nicheId: request.nicheId,
      status: 'pending'
    });
    
    // 2. Busca inteligência no MongoDB (Cérebro)
    const intelligence = await this.mongo.getIntelligence({
      niche: request.niche,
      platform: request.platform,
      recentPerformance: true
    });
    
    // 3. Monta prompt otimizado com aprendizado
    const optimizedPrompt = this.buildPrompt({
      transcription: request.transcription,
      learnings: intelligence.patterns,
      bestPractices: intelligence.topPerformers
    });
    
    // 4. Chama Claude API
    const aiResponse = await this.claudeAPI.generate(optimizedPrompt);
    
    // 5. Salva no cérebro (MongoDB)
    const mongoDoc = await this.mongo.saveGeneration({
      generationId: generation.id,
      prompt: optimizedPrompt,
      response: aiResponse,
      predictedEngagement: intelligence.predict(aiResponse),
      context: intelligence.patterns
    });
    
    // 6. Atualiza PostgreSQL com resultado
    await this.postgres.updateGeneration(generation.id, {
      status: 'completed',
      mongoContentId: mongoDoc._id,
      cost: this.calculateCost(aiResponse)
    });
    
    return {
      content: aiResponse,
      intelligence: intelligence.insights,
      prediction: mongoDoc.predictedEngagement
    };
  }
  
  // Loop de Feedback Automático
  async recordPerformance(contentId: string, metrics: Metrics) {
    // 1. Busca no PostgreSQL
    const generation = await this.postgres.findGeneration(contentId);
    
    // 2. Atualiza cérebro no MongoDB
    await this.mongo.updatePerformance(generation.mongoContentId, {
      actualMetrics: metrics,
      learningPoints: this.extractLearnings(metrics),
      patternsConfirmed: this.validatePatterns(metrics)
    });
    
    // 3. Retreina modelo se necessário
    if (this.shouldRetrain()) {
      await this.retrainModel();
    }
  }
}

      case 'fast':
        return this.useHaiku(request); // Mais rápido
    }
  }
  
  // Fallback automático
  async withFallback(primary, secondary) {
    try {
      return await primary();
    } catch (error) {
      console.log('Primary failed, using fallback');
      return await secondary();
    }
  }
}

// config/ai-providers.ts
export const AI_PROVIDERS = {
  openai: {
    models: {
      cheap: 'gpt-4o-mini',
      balanced: 'gpt-4o',
      premium: 'gpt-4-turbo'
    },
    cost: { input: 0.15, output: 0.60 }
  },
  anthropic: {
    models: {
      fast: 'claude-haiku-4-5',
      balanced: 'claude-sonnet-4-5',
      premium: 'claude-opus-4-5'
    },
    cost: { input: 3.00, output: 15.00 }
  }
};

// services/ai-orchestrator.ts
class AIOrchestrator {
  
  selectProvider(request: GenerationRequest) {
    // Lógica inteligente de seleção
    const complexity = this.analyzeComplexity(request);
    const budget = this.getUserBudget(request.userId);
    const priority = request.priority || 'balanced';
    
    if (priority === 'cost') {
      return { provider: 'openai', model: 'gpt-4o-mini' };
    }
    
    if (priority === 'quality' && budget > 0.20) {
      return { provider: 'anthropic', model: 'claude-opus-4-5' };
    }
    
    // Default: Claude Sonnet (melhor custo/benefício criativo)
    return { provider: 'anthropic', model: 'claude-sonnet-4-5' };
  }
  
  async generate(request: GenerationRequest) {
    const { provider, model } = this.selectProvider(request);
    
    try {
      // Tenta provider primário
      return await this.callProvider(provider, model, request);
    } catch (error) {
      // Fallback automático
      console.log(`${provider} failed, using fallback`);
      const fallback = provider === 'openai' ? 'anthropic' : 'openai';
      return await this.callProvider(fallback, 'balanced', request);
    }
  }
  
  // Cache inteligente
  async generateWithCache(request: GenerationRequest) {
    const cacheKey = this.generateCacheKey(request);
    
    // Verifica cache (MongoDB)
    const cached = await this.cache.get(cacheKey);
    if (cached && this.isCacheValid(cached)) {
      return { ...cached, fromCache: true, cost: 0 };
    }
    
    // Gera novo
    const result = await this.generate(request);
    
    // Salva cache
    await this.cache.set(cacheKey, result, { ttl: 3600 });
    
    return result;
  }
}
